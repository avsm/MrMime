let p_month p state =
  (Logs.debug @@ fun m -> m "state: p_month");

  let month = p_repeat ~a:3 ~b:3 is_alpha state in

  let month = match month with
  | "Jan" -> `Jan
  | "Feb" -> `Feb
  | "Mar" -> `Mar
  | "Apr" -> `Apr
  | "May" -> `May
  | "Jun" -> `Jun
  | "Jul" -> `Jul
  | "Aug" -> `Aug
  | "Sep" -> `Sep
  | "Oct" -> `Oct
  | "Nov" -> `Nov
  | "Dec" -> `Dec
  | str   -> raise (Error.Error (Error.err_unexpected_str str state)) in

  p month state
