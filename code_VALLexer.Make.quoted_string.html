let rec quoted_string buffer lexbuf = match%sedlex lexbuf with
      | qtext ->
        Buffer.add_string buffer (S.lexeme lexbuf);
        quoted_string buffer lexbuf
      | quoted_pair ->
        Buffer.add_string buffer (S.lexeme lexbuf);
        quoted_string buffer lexbuf
      | '"' ->
        Buffer.contents buffer
      | _ -> raise Lexical_error
