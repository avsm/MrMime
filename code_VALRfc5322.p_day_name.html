let p_day_name p state =
  (Logs.debug @@ fun m -> m "state: p_day_name");

  let day = p_repeat ~a:3 ~b:3 is_alpha state in

  let day = match day with
  | "Mon" -> `Mon
  | "Tue" -> `Tue
  | "Wed" -> `Wed
  | "Thu" -> `Thu
  | "Fri" -> `Fri
  | "Sat" -> `Sat
  | "Sun" -> `Sun
  | str   -> raise (Error.Error (Error.err_unexpected_str str state)) in

  p day state
