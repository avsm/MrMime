let p_decode stop p state = let buf = Buffer.create 16 in let rec decode state = let rec aux = function | `Read (buf, off, len, k) -> `Read (buf, off, len, (fun i -> aux @@ safe k i)) | #Error.err as err -> err | `Stop state -> p (Buffer.contents buf) state | `Continue state -> match cur_chr state with | '=' -> p_chr '=' state; if p_try is_hex_octet state = 2 then begin let s = p_repeat ~a:2 ~b:2 is_hex_octet state in F.add_char buf (F.hex s); decode state end else begin p_chr '\r' state; p_chr '\n' state; decode state end | '\x20' | '\x09' -> let lwsp = p_while Rfc822.is_lwsp state in if cur_chr state = '\r' then begin p_chr '\r' state; p_chr '\n' state; F.add_newline buf; decode state end else begin F.add_string buf lwsp; decode state end | chr when is_safe_char chr -> junk_chr state; F.add_char buf chr; decode state | chr -> raise (Error.Error (Error.err_unexpected chr state)) in aux (safe stop state) in decode state
